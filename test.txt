S = [];
K = [9, 3, 1, 1, 7];
N = 14;

def swap(i, j) {
    temp = S[i];
    S[i] = S[j];
    S[j] = temp;
}

def init() {
    i = 0;
    while i < N {
        push(S, i);
        i = i + 1;
    }
}

init();

i = 0;
j = 0;
while i < N {
    j = (j + S[i] + K[i % len(K)]) % N;
    swap(i, j);
    i = i + 1;
}

print(S);
